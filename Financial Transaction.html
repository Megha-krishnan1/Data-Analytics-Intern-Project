import os


os.environ.setdefault('OMP_NUM_THREADS', '1')

import warnings
warnings.filterwarnings('ignore') 

s
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from matplotlib.dates import DateFormatter
import seaborn as sns
from sklearn.preprocessing import StandardScaler
from sklearn.cluster import KMeans
from statsmodels.tsa.holtwinters import ExponentialSmoothing


os.makedirs('outputs/figures', exist_ok=True)

# ---------- 1. LOAD DATA ----------
file_path = r"C:\Users\HP\Downloads\Daily Household Transactions.csv"
print('Loading:', file_path)

try:
    df = pd.read_csv(file_path)
except Exception as e:
    raise SystemExit(f"Failed to load CSV. Check path and environment. Error: {e}")

print('Initial shape:', df.shape)
print(df.columns.tolist())

# ---------- 2. INITIAL CLEANING ----------
# Normalizing column names
df.columns = df.columns.str.strip().str.lower().str.replace(' ', '_')

# If 'date' col exists, parse it. If not, attempt to find a date-like column
if 'date' not in df.columns:
    raise SystemExit("No 'date' column found in the CSV. Please check column names.")

# Parsing date (dayfirst True to handle dd/mm/yyyy formats)
df['date'] = pd.to_datetime(df['date'], dayfirst=True, errors='coerce')

# Converting amount to numeric
if 'amount' not in df.columns:
    raise SystemExit("No 'amount' column found in the CSV. Please check column names.")

# Removing currency symbols and commas if present
if df['amount'].dtype == object:
    df['amount'] = df['amount'].astype(str).str.replace(r'[,$₹]', '', regex=True)

df['amount'] = pd.to_numeric(df['amount'], errors='coerce')

# Dropping rows missing these crucial fields
df = df.dropna(subset=['date', 'amount']).reset_index(drop=True)

print('Shape after dropna:', df.shape)

# ---------- 3. CLEANING & STANDARDIZATION (future-proof) ----------
# Ensuring string columns exist before manipulation
for col in ['mode', 'category', 'subcategory', 'income/expense', 'note', 'currency']:
    if col in df.columns:
        df[col] = df[col].astype(str).str.strip()

# Filling missing subcategory/note safely
if 'subcategory' in df.columns:
    df['subcategory'] = df['subcategory'].fillna('Unknown')
if 'note' in df.columns:
    df['note'] = df['note'].fillna('')

# Standardizing income/expense
if 'income/expense' in df.columns:
    df['type'] = df['income/expense'].str.lower().str.replace(' ', '_')
else:
    # Attempt to infer type if missing: if amount negative -> expense
    df['type'] = np.where(df['amount'] < 0, 'expense', 'income')

# Expense flag (safe with na=False)
df['is_expense'] = df['type'].str.contains('expens', na=False)

# ---------- 4. FEATURE ENGINEERING ----------
df['year'] = df['date'].dt.year
# Use 'ME' (month end) to be future-proof
df['month'] = df['date'].dt.to_period('M').astype(str)
df['day'] = df['date'].dt.date
df['weekday'] = df['date'].dt.day_name()
df['hour'] = df['date'].dt.hour

# Convert amounts so expenses are positive numbers for aggregation
# If your dataset records expenses as negative, decide how to treat them. We'll take absolute value for 'spend'
# Create a 'signed_amount' and a 'abs_amount' for clarity
if (df['amount'] < 0).any():
    df['signed_amount'] = df['amount']
    df['abs_amount'] = df['amount'].abs()
else:
    df['signed_amount'] = np.where(df['is_expense'], -df['amount'], df['amount'])
    df['abs_amount'] = df['amount']

# ---------- 5. SUMMARY METRICS & AGGREGATIONS ----------
total_spent = df.loc[df['is_expense'], 'abs_amount'].sum()
total_income = df.loc[~df['is_expense'], 'abs_amount'].sum()
net = total_income - total_spent
print(f"Total spent: {total_spent:.2f}, Total income: {total_income:.2f}, Net: {net:.2f}")

# Daily and monthly series
daily = df.set_index('date').resample('D')['signed_amount'].sum()
monthly = df.set_index('date').resample('ME')['signed_amount'].sum()  # ME = month end

# Top categories by spend (absolute)
if 'category' in df.columns:
    cat_sum = df.groupby('category')['abs_amount'].sum().sort_values(ascending=False)
else:
    cat_sum = pd.Series(dtype=float)

print('\nTop categories (if available):')
print(cat_sum.head(10))

# Save basic tables
cat_sum.to_csv('outputs/figures/top_categories.csv')
monthly.to_csv('outputs/figures/monthly_series.csv')

# ---------- 6. VISUALIZATIONS (EDA) ----------
plt.figure(figsize=(10,5))
plt.plot(monthly.index, monthly.values)
plt.title('Monthly Signed Amount (Income - Expense)')
plt.xlabel('Month')
plt.ylabel('Signed Amount')
plt.grid(True)
plt.tight_layout()
plt.savefig('outputs/figures/monthly_trend.png')
plt.show()

# Top categories bar chart
if not cat_sum.empty:
    plt.figure(figsize=(10,6))
    sns.barplot(x=cat_sum.values[:10], y=cat_sum.index[:10])
    plt.title('Top 10 Categories by Spend (absolute)')
    plt.xlabel('Total Spend')
    plt.tight_layout()
    plt.savefig('outputs/figures/top_categories.png')
    plt.show()

# Daily distribution (histogram of abs_amount)
plt.figure(figsize=(8,4))
plt.hist(df['abs_amount'].clip(upper=df['abs_amount'].quantile(0.99)), bins=50)
plt.title('Distribution of Transaction Amounts (clipped at 99th percentile)')
plt.xlabel('Amount')
plt.ylabel('Count')
plt.tight_layout()
plt.savefig('outputs/figures/amount_hist.png')
plt.show()

# Weekday spend boxplot
if 'weekday' in df.columns:
    pivot_wd = df.groupby('weekday')['abs_amount'].sum().reindex(
        ['Monday','Tuesday','Wednesday','Thursday','Friday','Saturday','Sunday'])
    plt.figure(figsize=(8,4))
    sns.barplot(x=pivot_wd.index, y=pivot_wd.values)
    plt.xticks(rotation=45)
    plt.title('Spend by Weekday')
    plt.tight_layout()
    plt.savefig('outputs/figures/weekday_spend.png')
    plt.show()

# ---------- 7. SMOOTHING & FORECAST (Holt-Winters) ----------
# We'll forecast the signed monthly series for the next 6 months
monthly_positive = monthly.copy()  # signed series; Holt-Winters accepts negative values too
# Fill tiny gaps
monthly_positive = monthly_positive.asfreq('ME').fillna(0)

try:
    model = ExponentialSmoothing(monthly_positive, seasonal='add', seasonal_periods=12).fit()
    forecast = model.forecast(6)
    # plot
    plt.figure(figsize=(10,5))
    plt.plot(monthly_positive.index, monthly_positive.values, label='Observed')
    plt.plot(forecast.index, forecast.values, label='Forecast', linestyle='--')
    plt.title('Holt-Winters Forecast (6 months)')
    plt.legend()
    plt.tight_layout()
    plt.savefig('outputs/figures/forecast_hw.png')
    plt.show()
except Exception as e:
    print('Forecasting failed:', e)

# ---------- 8. CATEGORY-LEVEL PIVOT (time x category) ----------
if 'category' in df.columns:
    pivot = df.pivot_table(index=pd.Grouper(key='date', freq='ME'),
                           columns='category', values='abs_amount', aggfunc='sum', fill_value=0)
    pivot.to_csv('outputs/figures/pivot_time_category.csv')

# ---------- 9. ANOMALY DETECTION (simple z-score on daily series) ----------
from scipy import stats
z = stats.zscore(daily.fillna(0))
anomalies = daily[~np.isfinite(z) | (np.abs(z) > 3)]
if not anomalies.empty:
    anomalies.to_csv('outputs/figures/daily_anomalies.csv')
    print('Anomalies detected and saved:', anomalies.shape[0])

# ---------- 10. CLUSTERING: KMeans on category patterns ----------
if 'category' in df.columns:
    # Create monthly category matrix
    mat = pivot.T.fillna(0)  # categories x months
    scaler = StandardScaler()
    mat_s = scaler.fit_transform(mat)

    kmeans = KMeans(n_clusters=3, random_state=42)
    labels = kmeans.fit_predict(mat_s)
    cluster_df = pd.DataFrame({'category': mat.index, 'cluster': labels})
    cluster_df.to_csv('outputs/figures/category_clusters.csv', index=False)
    print('Cluster assignments saved.')

# ---------- 11. SAVE CLEANED DATA ----------
df.to_csv('outputs/figures/cleaned_transactions.csv', index=False)

print('\nAll done. Outputs saved in outputs/figures/*.')
